print('Estes são os resultados referentes ao valor {}:'.format(valor))
print('O dobro de {} é {}.'.format(valor, valor*2))
print('O triplo de {} é {}.'.format(valor, valor*3))
print('A raíz quadrada de {} é {:.2f}.'.format(valor, valor**(1/2)))
